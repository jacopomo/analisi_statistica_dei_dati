\subsubsection{Binomial: For fixed trials}
Support: $k \in \{0, 1, \dots, n\}$ and $p \in [0, 1]$.

The probability of having exactly $k$ successes in $n$ independent Bernoulli trials is:

\begin{equation}
	\mathcal{B}_n(k;p) = \binom{n}{k}p^k\left(1-p\right)^{n-k}
\tag{Binomial pmf}
\label{pmf:binom}
\end{equation}

The Moment Generating Function (MGF) is the most efficient way to derive the raw moments:

	\[
	M_X(t) = \mathbb{E}[e^{tX}] = (1 - p + pe^t)^n
	\]

Which brings us to the mean and variance of $k$:

	\[
	\E\left[k\right] = np \quad \Var\left[k\right] = np(1-p)
	\]

The log-Likelihood for a sample of size $m$ (where each observation $k_i$ is the number of successes in $n$ trials) is:

	\[
	\ln L(p) = \sum_{i=1}^m \left[ \ln\binom{n}{k_i} + k_i\ln p + (n-k_i)\ln(1-p) \right]
	\]

which brings the MLE for $p$:

	\[
	\hat{p}_{MLE} = \frac{1}{mn} \sum_{i=1}^m k_i = \frac{\bar{k}}{n}
	\]

Which is an unbiased estimator with variance:

	\[
	\Var\left[\hat{p}_{MLE}\right] = \frac{1}{n^2}\Var[k] = \frac{np(1-p)}{n^2} = \frac{p(1-p)}{n}
	\]


Fisher Information: This measure tells us how much "information" $k$ carries about the unknown parameter $p$. For a single observation:

	\[
	\mathcal{I}(p) = -\E\left[ \frac{\partial^2 \ell}{\partial p^2} \right] = \frac{n}{p(1-p)}
	\] 

This shows us at a glance that $\hat{p}_{MLE}$ is the minimum variance estimator.

\subsubsection{Poisson: For occurrences over time/space}

\label{eq:poisson_mle}
We now want to find the maximum likelihood estimator (MLE) for a Poisson distribution. 
Given a sample of $N$ independent observations $\{x_i\}$, the total likelihood is:
\begin{align*}
	L_{\text{tot}}(\mu) &= \prod_{i=1}^N \frac{\mu^{x_i} e^{-\mu}}{x_i!}
\end{align*}
Taking the logarithm, we obtain the log-likelihood:
\begin{align*}
	\ln L_{\text{tot}}(\mu) 
	&= \sum_{i=1}^N \ln \left( \frac{\mu^{x_i} e^{-\mu}}{x_i!} \right) \\
	&= \sum_{i=1}^N x_i \ln \mu - \sum_{i=1}^N \mu - \sum_{i=1}^N \ln(x_i!)
\end{align*}
Since the last term does not depend on $\mu$, it can be neglected. Therefore,
\[
\ln L_{\text{tot}}(\mu) = \sum_{i=1}^N x_i \ln \mu - N\mu
\]

To find the MLE, we solve the \ref{eq:likelihood}:
\[
\frac{\partial \ln L_{\text{tot}}(\mu)}{\partial \mu} 
= \sum_{i=1}^N \frac{x_i}{\mu} - N = 0
\]
Evaluating at $\mu = \hat{\mu}$, we obtain:
\begin{equation*}
	\hat{\mu} = \frac{1}{N} \sum_{i=1}^N x_i
	\label{eq:poissonian_mle}
	\tag{Poisson MLE}
\end{equation*}

As expected, the MLE for the Poisson mean coincides with the sample mean. In the simple case of a single observation of $n$ events, we trivially get: $\hat\mu=n$.

\subsubsection{Uniform: The baseline for probability}

The general Uniform distribution $\mathcal{U}[l, u]$ with $l, u \in \mathbb{R}$ has the \emph{pdf}:
\[
p(x; l, u)=
\begin{cases}
	\frac{1}{u-l} & \text{if } x \in [l, u] \\
	0 & \text{otherwise}
\end{cases}
\]

One can equivalently parameterize the distribution in terms of the interval length $\delta$ and its center $\mu$:
\[
p(x; \delta, \mu)=
\begin{cases}
	\frac{1}{\delta} & \text{if } x \in [\mu-\delta/2, \mu+\delta/2] \\
	0 & \text{otherwise}
\end{cases}
\]

To simplify the following derivations, we consider a Uniform $\mathcal{U}[0, m]$ (any other case can be reduced to this one through a simple change of variables).

The expected value is:
\begin{align*}
	\E[x]
	&=\int_{-\infty}^{+\infty} x\,p(x;0,m)\,dx
	=\int_{0}^{m} x\frac{1}{m}\,dx
	=\frac{m}{2}
\end{align*}

and the variance:
\begin{align*}
	\operatorname{Var}[x]
	&=\E[x^2]-\E[x]^2 \\
	&=\int_{0}^{m} x^2\frac{1}{m}\,dx - \frac{m^2}{4} \\
	&=\frac{m^2}{3}-\frac{m^2}{4}
	=\frac{m^2}{12}
\end{align*}

Taking the square root, we arrive to the very familiar standard deviation of a uniformly distributed variable:
	\[
	\sigma = \frac{m}{\sqrt{12}}
	\]

This is why when we measure something up to a certain sensitivity (for example, the ticks on a ruler), we often divide the minimum interval by $\sqrt{12}$ and use that as the statistical error. We are essentially assuming the "true" value is uniformly distributed within that interval. 

The characteristic function is:
\[
\phi(t)=\E[e^{itx}]
=\frac{e^{imt}-1}{imt}
\]

We now compute the likelihood $L_x(m)$:
\[
L_x(m)=\frac{1}{m}\,\mathbb{I}_{\{m \geq x\}}
\]
(see \ref{ex:uniform_likelihood}).

The Fisher information is:
\[
I(m)=\E \left[\left(\frac{\partial \ln L_x(m)}{\partial m}\right)^2 \right]
=\frac{1}{m^2}
\]

Given $N$ observations $\{x_i\}$ drawn independently from a uniform distribution, it is often useful to determine the distributions of 
\[
x_{\max} = \max_i [x_i]
\quad \text{and} \quad
x_{\min} = \min_i [x_i]
\]

We start with $x_{\max}$. Its distribution can be derived in two equivalent ways:

\begin{itemize}
	\item Using the cumulative distribution function:
	\[
	F_{x_{\max}}(x)=P(x_{\max}\leq x)
	\]
	The condition $x_{\max}\leq x$ means that all observations satisfy $x_i \leq x$, hence:
	\[
	P(x_{\max}\leq x)=P(x_1\leq x, \dots, x_N \leq x)
	=\prod_{i=1}^{N}P(x_i\leq x)
	\]
	where we used independence.
	
	For the uniform distribution:
	\[
	P(x_i\leq x)=
	\begin{cases}
		0 & \text{if } x<0\\
		\frac{x}{m} & \text{if } 0\leq x \leq m\\
		1 & \text{if } x> m
	\end{cases}
	\]
	
	Thus, for $x\in [0,m]$:
	\[
	F_{x_{\max}}(x)=\left(\frac{x}{m}\right)^N
	\]
	
	Differentiating:
	\[
	p_{x_{\max}}(x)=\frac{d}{dx}F_{x_{\max}}(x)
	=
	\begin{cases}
		\frac{N}{m}\left(\frac{x}{m}\right)^{N-1} & \text{if } 0 \leq x \leq m\\
		0 & \text{otherwise}
	\end{cases}
	\]
	
	\item Another way to arrive at the same result is to view the problem in terms of a multinomial experiment. 
	
	We consider $N$ independent draws and imagine ordering them. Fix a value $x$ and focus on the $k$-th order statistic $x_k$. Each observation can fall into one of three categories:
	\begin{itemize}
		\item it lies below $x$
		\item it falls in an infinitesimal interval $[x, x+dx]$
		\item it lies above $x$
	\end{itemize}
	
	These three outcomes have probabilities:
	\[
	p_{<} = F(x), \qquad
	p_{=} = p(x)\,dx, \qquad
	p_{>} = 1 - F(x)
	\]
	
	For $N$ trials, the probability that exactly $k-1$ points lie below $x$, one point falls in $[x, x+dx]$, and the remaining $N-k$ lie above $x$ is given by the multinomial distribution:
	\[
	\frac{N!}{(k-1)!(N-k)!1!}
	p_{<}^{\,k-1}\,
	p_{=}\,
	p_{>}^{\,N-k}
	\]
	
	Dividing by $dx$, we obtain the density of the $k$-th order statistic:
	\[
	p_{x_k}(x)
	=
	\frac{N!}{(k-1)!(N-k)!}
	[F(x)]^{k-1}
	[1-F(x)]^{N-k}
	p(x)
	\]
	
	In the case of the maximum, we set $k=N$. Then there are $N-1$ points below $x$ and none above, giving:
	\begin{align*}
		p_{x_{\max}}(x)
		&= N [F(x)]^{N-1} p(x) \\
		&= N \left(\frac{x}{m}\right)^{N-1}\frac{1}{m} \\
		&= \frac{N}{m}\left(\frac{x}{m}\right)^{N-1}
	\end{align*}
	
	which coincides with the result obtained via the cumulative distribution.
\end{itemize}

We now turn to $x_{\min}$. Its distribution can be derived in analogous ways:

\begin{itemize}
	\item Using the cumulative distribution function:
	\[
	F_{x_{\min}}(x)=P(x_{\min}\leq x)
	\]
	It is often more convenient to compute the complementary probability:
	\[
	P(x_{\min} > x)
	\]
	The condition $x_{\min} > x$ means that all observations satisfy $x_i > x$, hence:
	\[
	P(x_{\min} > x)=P(x_1 > x, \dots, x_N > x)
	=\prod_{i=1}^{N}P(x_i > x)
	\]
	
	For the uniform distribution:
	\[
	P(x_i > x)=
	\begin{cases}
		1 & \text{if } x<0\\
		1-\frac{x}{m} & \text{if } 0\leq x \leq m\\
		0 & \text{if } x> m
	\end{cases}
	\]
	
	Thus, for $x\in [0,m]$:
	\[
	P(x_{\min} > x)=\left(1-\frac{x}{m}\right)^N
	\]
	
	and therefore:
	\[
	F_{x_{\min}}(x)=1-\left(1-\frac{x}{m}\right)^N
	\]
	
	Differentiating:
	\[
	p_{x_{\min}}(x)=\frac{d}{dx}F_{x_{\min}}(x)
	=
	\begin{cases}
		\frac{N}{m}\left(1-\frac{x}{m}\right)^{N-1} & \text{if } 0 \leq x \leq m\\
		0 & \text{otherwise}
	\end{cases}
	\]
	
	\item Using the multinomial argument:
	
	We proceed as before by fixing a value $x$ and considering the $k$-th order statistic $x_k$.
	The density of $x_k$ is:
	\[
	p_{x_k}(x)
	=
	\frac{N!}{(k-1)!(N-k)!}
	[F(x)]^{k-1}
	[1-F(x)]^{N-k}
	p(x)
	\]
	
	For the minimum, we set $k=1$. Then there are no points below $x$ and $N-1$ points above it:
	\begin{align*}
		p_{x_{\min}}(x)
		&= N [1-F(x)]^{N-1} p(x) \\
		&= N \left(1-\frac{x}{m}\right)^{N-1}\frac{1}{m} \\
		&= \frac{N}{m}\left(1-\frac{x}{m}\right)^{N-1}
	\end{align*}
	
	which agrees with the previous result.
	
	\item Variable change from $x_{\max}$:
	
	Let $y_i = m - x_i$. Then each $y_i \sim \mathcal{U}[0,m]$ and:
	\[
	y_{\max} = m - x_{\min}
	\]
	
	Since we already know the distribution of $y_{\max}$, we can obtain that of $x_{\min}$ by a change of variables:
	\[
	p_{x_{\min}}(x)
	=
	p_{y_{\max}}(m-x)
	\left|\frac{d}{dx}(m-x)\right|
	\]
	
	which again yields:
	\[
	p_{x_{\min}}(x)
	=
	\frac{N}{m}\left(1-\frac{x}{m}\right)^{N-1}
	\]
\end{itemize}

While we are at it, we can also find expected value and variance for $x_{\max}$ and $x_{\min}$:
\begin{align*}
	\E_{x_{\max}}[x]&=\int_{0}^{m}x\frac{N}{m}\left(\frac{x}{m}\right)^{N-1}\;dx=\frac{N}{m^N}\int_{0}^{m}x^N\;dx\\
	&=\frac{N}{m^N}\left[\frac{1}{N+1}x^{N+1}\right]^m_0=m \frac{N}{N+1}\\ \\
	\E_{x_{\min}}[x]&=\int_{0}^{m}x\frac{N}{m}\left(1-\frac{x}{m}\right)^{N-1}\;dx\\
	&=\frac{N}{m}\left[-\frac{m}{N}x\left(1-\frac{x}{m}\right)^{N}\bigg\vert_0^m-\int_0^m-\frac{m}{N}\left(1-\frac{x}{m}\right)^{N}\;dx\right]\\
	&=\int_0^m\left(1-\frac{x}{m}\right)^N\;dx=-\frac{m}{N+1}\left(1-\frac{x}{m}\right)^{N+1}\bigg\vert_0^m\\
	&=m\frac{1}{N+1}
\end{align*}
